from copy import deepcopy
from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData

from pkdb_models.models import sorafenib

from pkdb_models.models.sorafenib.experiments.base_experiment import (
    SorafenibSimulationExperiment
)
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.sorafenib.helpers import run_experiments


class Awada2005(SorafenibSimulationExperiment):
    """Simulation experiment of Awada2005."""

    doses = [100, 200, 300, 400, 600, 800]

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        for fig_id in ["Fig2"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)

            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                dset.unit_conversion("mean", 1 / self.Mr.sor)
                dsets[label] = dset

        # print(dsets.keys())
        # print(dsets)

        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        for dose in self.doses:
            tc0 = Timecourse(
                start=0,
                end=24 * 60,  # [min]
                steps=200,
                changes={
                    **self.default_changes(),
                    "PODOSE_sor": Q_(dose, "mg"),
                },
            )
            tc1 = Timecourse(
                start=0,
                end=24 * 60,  # [min]
                steps=200,
                changes={
                    "PODOSE_sor": Q_(dose, "mg"),
                },
            )
            tc2 = Timecourse(
                start=0,
                end=100 * 60,  # [min]
                steps=200,
                changes={
                    "PODOSE_sor": Q_(dose, "mg"),
                },
            )

            tcsims[f"sor_po{dose}"] = TimecourseSim(
                [tc0] + [deepcopy(tc1) for _ in range(19)] + [tc2],
            )

        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for dose in self.doses:
            for day in [1, 7, 21]:
                dataset = f"SOF{dose}_D{day}"
                mappings[f"fm_sor_po_{dataset}"] = FitMapping(
                    self,
                    reference=FitData(
                        self,
                        dataset=dataset,
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                    ),
                    observable=FitData(
                        self, task=f"task_sor_po{dose}", xid="time", yid="[Cve_sor]",
                    ),
                )
        return mappings

    def figures(self) -> Dict[str, Figure]:
        name = "Fig2"
        fig = Figure(
            experiment=self,
            sid=f"{name}_PO_plasma",
            num_rows=4,
            num_cols=1,
            name=f"{self.__class__.__name__}",
        )

        plots = fig.create_plots(
            xaxis=Axis(self.label_time, unit="hour"),
            yaxis=Axis(self.label_sor, unit=self.units["[Cve_sor]"]),
            legend=False
        )
        plots[1].xaxis.min = -1
        plots[1].xaxis.max = 24 + 1

        plots[2].xaxis.min = 24 * 6 - 1
        plots[2].xaxis.max = 24 * 7 + 1

        plots[3].xaxis.min = 24 * 20 - 1
        plots[3].xaxis.max = 24 * 20 + 96 + 1

        # simulation
        for dose in self.doses:
            for k in [0, 1, 2, 3]:
                plots[k].add_data(
                    task=f"task_sor_po{dose}",
                    xid="time",
                    yid="[Cve_sor]",
                    label=f"Sim{dose} [mg]",
                    color=self.color_for_dose(dose),
                )

        # data
        for dose in self.doses:
            for day in [1, 7, 21]:
                if day == 1:
                    ks = [0, 1]
                elif day == 7:
                    ks = [0, 2]
                elif day == 21:
                    ks = [0, 3]

                for k in ks:
                    dataset = f"SOF{dose}_D{day}"
                    plots[k].add_data(
                        dataset=dataset,
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                        label=dataset,
                        color=self.color_for_dose(dose),
                    )

        return {fig.sid: fig}


if __name__ == "__main__":
    out = sorafenib.RESULTS_PATH_SIMULATION / Awada2005.__name__
    out.mkdir(parents=True, exist_ok=True)
    run_experiments(Awada2005, output_dir="Awada2005")
